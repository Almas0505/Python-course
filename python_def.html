# Дәстүрлі функция def square(x): return x ** 2 # Lambda функциясы square_lambda = lambda x: x ** 2 print(square(5)) # 25 print(square_lambda(5)) # 25 # Lambda-ны map, filter-мен пайдалану numbers = [1, 2, 3, 4, 5] squares = list(map(lambda x: x**2, numbers)) print(squares) # [1, 4, 9, 16, 25]

🚀 Күрделі Функция Мысалдары

1. Рекурсивті функциялар

def factorial(n): """n! есептейді (рекурсия арқылы)""" if n <= 1: return 1 else: return n * factorial(n - 1) print(factorial(5)) # 120 (5*4*3*2*1) def fibonacci(n): """Фибоначчи сандары""" if n <= 1: return n return fibonacci(n-1) + fibonacci(n-2) print([fibonacci(i) for i in range(8)]) # [0, 1, 1, 2, 3, 5, 8, 13]

2. *args және **kwargs

def flexible_function(*args, **kwargs): """Кез келген санда аргументтерді қабылдайды""" print(f"Позициялық аргументтер: {args}") print(f"Кілт сөзді аргументтер: {kwargs}") # Барлық сандарды қосу total = sum(args) return total # Әртүрлі шақырулар print(flexible_function(1, 2, 3)) print(flexible_function(1, 2, name="Айгүл", age=25))

🎯 Практикалық Жоба: Кітапхана Жүйесі

# Кітапхана басқару жүйесі books_database = [] def add_book(title, author, year, genre="Белгісіз"): """Жаңа кітап қосу""" book = { "id": len(books_database) + 1, "title": title, "author": author, "year": year, "genre": genre, "available": True } books_database.append(book) return f"'{title}' кітабы қосылды!" def search_books(search_term, search_by="title"): """Кітаптарды іздеу""" results = [] for book in books_database: if search_term.lower() in book[search_by].lower(): results.append(book) return results def get_statistics(): """Кітапхана статистикасы""" total_books = len(books_database) available_books = sum(1 for book in books_database if book["available"]) borrowed_books = total_books - available_books return { "жалпы_кітаптар": total_books, "қолжетімді": available_books, "алынған": borrowed_books }

🧮 Пайдалану мысалы

# Кітаптар қосу print(add_book("Абай жолы", "Мұхтар Әуезов", 1942, "Роман")) print(add_book("Қан мен тер", "Әбдіжәміл Нұрпейісов", 1961)) # Іздеу search_results = search_books("Абай") print(f"Табылған кітаптар: {len(search_results)}") # Статистика stats = get_statistics() print(f"Статистика: {stats}")

Келесі функцияның нәтижесі қандай болады?

def mystery_function(x, y=2): return x ** y result = mystery_function(3) + mystery_function(2, 3) print(result)
А) 11
Б) 13
В) 17
Г) 25
Дұрыс жауап: В) 17
Себебі: mystery_function(3) = 3² = 9, mystery_function(2, 3) = 2³ = 8, 9 + 8 = 17
